const MONTHS_FR = ['Janvier','Février','Mars','Avril','Mai','Juin','Juillet','Août','Septembre','Octobre','Novembre','Décembre'];

function toIso(d) { return d.getFullYear()+'-'+String(d.getMonth()+1).padStart(2,'0')+'-'+String(d.getDate()).padStart(2,'0'); }

function parseDate(s) { return s ? new Date(s+'T00:00:00') : null; }

function genId()  { return 'c'+Date.now()+Math.random().toString(36).slice(2,6); }

function escHtml(s) { return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;'); }

function escAttr(s) { return String(s||'').replace(/"/g,'&quot;'); }

function statClass(s)  { return ({Brouillon:'s-brouillon',Validé:'s-valide',Envoyé:'s-envoye'})[s]||''; }

function cibleClass(s) { return ({RC:'t-rc',Cellules:'t-cellules','RC + Cellules':'t-rcc'})[s]||''; }

function canalClass(s) { return ({Email:'c-email',Teams:'c-teams',Réunion:'c-reunion',Newsletter:'c-newsletter',SMS:'c-sms',Courrier:'c-courrier'})[s]||''; }

function canalEmoji(s) { return ({Email:'📧',Teams:'💬',Réunion:'🤝',Newsletter:'📰',SMS:'📱',Courrier:'✉️'})[s]||''; }

function typeEmoji(s)  { return ({Information:'ℹ️','Action requise':'⚡',Rappel:'🔔',Invitation:'🎯','Compte-rendu':'📋'})[s]||''; }

function filteredComms() { return comms.filter(c => Object.entries(filters).every(([k,v]) => !v||c[k]===v)); }

/* ─── JSONBIN ─── */

async function fetchComms() {

  // Public read — no auth needed if bin is public

  const headers = {'X-Bin-Meta':'false'};

  if (adminKey) headers['X-Master-Key'] = adminKey;

  const res = await fetch(BIN_URL+'/latest', { headers });

  if (!res.ok) throw new Error('HTTP '+res.status);

  const json = await res.json();

  return Array.isArray(json.comms) ? json.comms : [];

}


async function pushComms(arr) {

  if (!adminKey) throw new Error('No admin key');

  const res = await fetch(BIN_URL, {

    method: 'PUT',

    headers: { 'X-Master-Key': adminKey, 'Content-Type': 'application/json' },

    body: JSON.stringify({ comms: arr })

  });

  if (!res.ok) throw new Error('HTTP '+res.status);

}


async function loadData() {

  setSyncStatus('syncing', 'Chargement…');

  try {

    comms = await fetchComms();

    setSyncStatus('synced', 'Mis à jour · '+new Date().toLocaleTimeString('fr-FR',{hour:'2-digit',minute:'2-digit'}));

    render();

  } catch(e) {

    setSyncStatus('error', 'Erreur de connexion');

    console.error(e);

  }

}


async function saveData() {

  setSyncStatus('syncing', 'Sauvegarde…');

  try {

    await pushComms(comms);

    setSyncStatus('synced', 'Sauvegardé · '+new Date().toLocaleTimeString('fr-FR',{hour:'2-digit',minute:'2-digit'}));

    showToast('✓ Sauvegardé');

  } catch(e) {

    setSyncStatus('error', 'Erreur de sauvegarde');

    showToast('⚠ Erreur — modifications non sauvegardées');

    console.error(e);

  }

}


function startAutoRefresh() {

  if (autoTimer) clearInterval(autoTimer);

  autoTimer = setInterval(loadData, 60000);

}

/* MONTH */

function renderMonth() {

  const area = document.getElementById('calendar-area');

  const fc = filteredComms();

  const byDate = {};

  fc.forEach(c => { if(!c.date)return; (byDate[c.date]=byDate[c.date]||[]).push(c); });

  const year=cursor.getFullYear(), month=cursor.getMonth();

  const first=new Date(year,month,1), last=new Date(year,month+1,0);

  const startDow=(first.getDay()+6)%7;

  let days=[];

  for(let i=0;i<startDow;i++) days.push({date:new Date(year,month,1-startDow+i),other:true});

  for(let d=1;d<=last.getDate();d++) days.push({date:new Date(year,month,d),other:false});

  while(days.length%7!==0||days.length<42) days.push({date:new Date(year,month+1,days.length-startDow-last.getDate()+1),other:true});

  days=days.slice(0,42);


  let html='';


  // viewer banner

  if (!isAdmin) {

    html += `<div class="viewer-banner">👁 Mode lecture — seul l'administrateur peut modifier ce calendrier. Il se met à jour automatiquement.</div>`;

  }


  html+='<div class="month-grid">';

  DAYS_FR.forEach(d=>{html+=`<div class="day-header">${d}</div>`;});

  days.forEach(({date,other})=>{

    const iso=toIso(date), isToday=iso===toIso(today);

    const dc=byDate[iso]||[], MAX=3;

    html+=`<div class="day-cell${other?' other-month':''}${isToday?' today':''}${isAdmin?' clickable':''}" data-date="${iso}">`;

    html+=`<div class="day-num">${date.getDate()}</div>`;

    dc.slice(0,MAX).forEach(c=>{

      html+=`<div class="comm-pill ${statClass(c.statut)} ${cibleClass(c.cible)}" data-id="${c.id}" title="${escAttr(c.titre)}">${canalEmoji(c.canal)} ${escHtml(c.titre)}</div>`;

    });

    if(dc.length>MAX) html+=`<div class="more-badge">+${dc.length-MAX} autre${dc.length-MAX>1?'s':''}</div>`;

    html+='</div>';

  });

  html+='</div>';

  area.innerHTML=html;


  area.querySelectorAll('.day-cell.clickable').forEach(el=>{

    el.addEventListener('click',e=>{

      if(e.target.classList.contains('comm-pill')||e.target.classList.contains('more-badge'))return;

      if(isAdmin) openModal(null,el.dataset.date);

    });

  });

  area.querySelectorAll('.comm-pill').forEach(el=>{

    el.addEventListener('click',e=>{e.stopPropagation();openDetail(el.dataset.id);});

  });

}


/* WEEK */

function weekBounds(d) {

  const start=new Date(d), dow=(start.getDay()+6)%7;

  start.setDate(start.getDate()-dow);

  const end=new Date(start); end.setDate(end.getDate()+6);

  return {start,end};

}


function renderWeek() {

  const area=document.getElementById('calendar-area');

  const fc=filteredComms();

  const {start}=weekBounds(cursor);

  const days=[];

  for(let i=0;i<7;i++){const d=new Date(start);d.setDate(d.getDate()+i);days.push(d);}

  const byDate={};

  fc.forEach(c=>{if(!c.date)return;(byDate[c.date]=byDate[c.date]||[]).push(c);});


  let html='';

  if(!isAdmin) html+=`<div class="viewer-banner">👁 Mode lecture — seul l'administrateur peut modifier ce calendrier.</div>`;

  html+='<div class="week-grid">';

  html+='<div class="week-header" style="background:var(--surface);border-bottom:1px solid var(--border);"></div>';

  days.forEach(d=>{

    const isToday=toIso(d)===toIso(today);

    html+=`<div class="week-header${isToday?' today-col':''}" style="background:var(--surface);border-bottom:1px solid var(--border);">

      <div class="wday">${DAYS_FR[(d.getDay()+6)%7]}</div>

      <div class="wdate">${d.getDate()}</div>

    </div>`;

  html+='<div style="background:var(--surface);border-top:1px solid var(--border);padding:10px 8px;font-size:10px;font-weight:700;color:var(--muted);text-align:right;">COMMS</div>';

  days.forEach(d=>{

    const iso=toIso(d), dc=byDate[iso]||[];

    html+=`<div class="week-allday-cell${isAdmin?' clickable':''}" data-date="${iso}" style="${toIso(d)===toIso(today)?'background:#f8faff':''}">`;

    if(dc.length===0&&isAdmin) html+=`<div style="font-size:10px;color:#e2e8f0;padding:4px;">Ajouter…</div>`;

async function saveComm() {

  if(!isAdmin)return;

  const titre=document.getElementById('f-titre').value.trim();

  const date=document.getElementById('f-date').value;

  if(!titre){document.getElementById('f-titre').style.borderColor='#ef4444';document.getElementById('f-titre').focus();return;}

  if(!date){document.getElementById('f-date').style.borderColor='#ef4444';document.getElementById('f-date').focus();return;}

  const obj={

    id:editingId||genId(), titre, date,

    statut:document.getElementById('f-statut').value,

    cible:document.getElementById('f-cible').value,

    canal:document.getElementById('f-canal').value,

    type:document.getElementById('f-type').value,

    description:document.getElementById('f-desc').value.trim(),

    createdAt:editingId?(comms.find(c=>c.id===editingId)?.createdAt||new Date().toISOString()):new Date().toISOString(),

    updatedAt:new Date().toISOString(),

  };

  if(editingId){const idx=comms.findIndex(c=>c.id===editingId);if(idx>-1)comms[idx]=obj;}

  else comms.push(obj);

  const wasEditing=editingId;

  closeModal(); render();

  if(wasEditing) openDetail(obj.id);

  await saveData();

}

document.querySelectorAll('.filter-chip[data-filter]').forEach(chip=>{

  chip.addEventListener('click',()=>{

    const key=chip.dataset.filter, val=chip.dataset.val;

    if(filters[key]===val){delete filters[key];chip.classList.remove('active');}

    else{document.querySelectorAll(`.filter-chip[data-filter="${key}"]`).forEach(c=>c.classList.remove('active'));filters[key]=val;chip.classList.add('active');}

    render();

  });

});

document.getElementById('clear-filters').addEventListener('click',()=>{

  filters={};document.querySelectorAll('.filter-chip[data-filter]').forEach(c=>c.classList.remove('active'));render();

});


/* ─── EXPORT CSV ─── */

document.getElementById('export-btn').addEventListener('click',()=>{

  if(!comms.length){showToast('Aucune communication à exporter.');return;}

  const rows=[['ID','Titre','Date','Statut','Cible','Canal','Type','Description','Créé le','Modifié le']];

  comms.forEach(c=>rows.push([c.id,c.titre,c.date,c.statut,c.cible,c.canal,c.type,

    (c.description||'').replace(/"/g,'""'),

    c.createdAt?new Date(c.createdAt).toLocaleDateString('fr-FR'):'',

    c.updatedAt?new Date(c.updatedAt).toLocaleDateString('fr-FR'):'']));

  const csv=rows.map(r=>r.map(f=>`"${f||''}"`).join(';')).join('\n');

  const blob=new Blob(['\uFEFF'+csv],{type:'text/csv;charset=utf-8;'});

  const url=URL.createObjectURL(blob);

  const a=document.createElement('a');a.href=url;a.download='stream_comms_'+toIso(today)+'.csv';

  document.body.appendChild(a);a.click();document.body.removeChild(a);URL.revokeObjectURL(url);

});
